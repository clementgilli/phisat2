CLI usage

python -m phisat2.cli.train fit \
  --task segmentation \
  --dataset clouds \
  --model phisat2_geoaware \
  --dataloader synthetic \
  --seeds 0 1 \
  --root-dir . \
  --output-dir runs \
  --max-epochs 5 \
  --batch-size 8 \
  --crop-size 224 \
  --lr 1e-4 \
  --num-workers 4 \
  --accelerator cpu \
  --devices 1 \
  --precision 32-true \
  --no-pretrained
